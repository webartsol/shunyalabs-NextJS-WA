Schema Elements

Dysfluency Elements

  • <saml:repetition> - Repeated sounds/syllables
  • <saml:prolongation> - Prolonged sounds
  • <saml:block> - Speech blocks
  • <saml:unit> - Repeated unit

Transcription Elements

  • <saml:unintelligible> - Unclear speech
  • <saml:overlap> - Overlapping speech
  • <saml:letter> - Spelled letters

Sound Elements

  • <saml:nonspeech> - Speaker sounds
  • <saml:background> - Background noise
  • <saml:metadata> - File/speaker info

<?xml version="1.0" encoding="UTF-8"?>
<!--
    SAML - Stuttering Annotation Markup Language
    XML Schema Definition (XSD) v1.0
    
    Namespace: https://shunyalabs.ai/2026/saml
    
    Copyright (c) 2026 Shunya Labs (https://shunyalabs.ai)
    SPDX-License-Identifier: W3C-20150513
-->
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema"
           xmlns:saml="https://shunyalabs.ai/2026/saml"
           targetNamespace="https://shunyalabs.ai/2026/saml"
           elementFormDefault="qualified"
           version="1.0">

    <!-- ================================================================== -->
    <!-- SIMPLE TYPES                                                       -->
    <!-- ================================================================== -->

    <!-- Severity type: 0-8 based on duration -->
    <xs:simpleType name="severityType">
        <xs:restriction base="xs:integer">
            <xs:minInclusive value="0"/>
            <xs:maxInclusive value="8"/>
        </xs:restriction>
    </xs:simpleType>

    <!-- Duration in milliseconds -->
    <xs:simpleType name="durationType">
        <xs:restriction base="xs:positiveInteger"/>
    </xs:simpleType>

    <!-- Repetition types -->
    <xs:simpleType name="repetitionType">
        <xs:restriction base="xs:string">
            <xs:enumeration value="incomplete-syllable"/>
            <xs:enumeration value="syllable"/>
            <xs:enumeration value="mixed"/>
        </xs:restriction>
    </xs:simpleType>

    <!-- Block sound types -->
    <xs:simpleType name="blockSoundType">
        <xs:restriction base="xs:string">
            <xs:enumeration value="consonant"/>
            <xs:enumeration value="vowel"/>
        </xs:restriction>
    </xs:simpleType>

    <!-- Block audibility types -->
    <xs:simpleType name="blockAudibilityType">
        <xs:restriction base="xs:string">
            <xs:enumeration value="audible"/>
            <xs:enumeration value="inaudible"/>
        </xs:restriction>
    </xs:simpleType>

    <!-- Position type for prolongations -->
    <xs:simpleType name="positionType">
        <xs:restriction base="xs:string">
            <xs:enumeration value="initial"/>
            <xs:enumeration value="medial"/>
            <xs:enumeration value="final"/>
        </xs:restriction>
    </xs:simpleType>

    <!-- ================================================================== -->
    <!-- CORE DYSFLUENCY ELEMENTS                                           -->
    <!-- ================================================================== -->

    <!-- Repetition element -->
    <xs:element name="repetition">
        <xs:complexType>
            <xs:sequence>
                <xs:element ref="saml:unit" minOccurs="2" maxOccurs="unbounded"/>
            </xs:sequence>
            <xs:attribute name="type" type="saml:repetitionType" use="required"/>
            <xs:attribute name="severity" type="saml:severityType" use="required"/>
            <xs:attribute name="duration" type="saml:durationType" use="optional"/>
            <xs:attribute name="count" type="xs:positiveInteger" use="optional"/>
        </xs:complexType>
    </xs:element>

    <!-- Prolongation element -->
    <xs:element name="prolongation">
        <xs:complexType mixed="true">
            <xs:attribute name="sound" type="xs:string" use="required"/>
            <xs:attribute name="severity" type="saml:severityType" use="required"/>
            <xs:attribute name="duration" type="saml:durationType" use="optional"/>
            <xs:attribute name="position" type="saml:positionType" use="optional"/>
        </xs:complexType>
    </xs:element>

    <!-- Block element -->
    <xs:element name="block">
        <xs:complexType>
            <xs:attribute name="type" type="saml:blockAudibilityType" use="required"/>
            <xs:attribute name="sound" type="saml:blockSoundType" use="required"/>
            <xs:attribute name="severity" type="saml:severityType" use="required"/>
            <xs:attribute name="duration" type="saml:durationType" use="optional"/>
        </xs:complexType>
    </xs:element>

    <!-- Unit element -->
    <xs:element name="unit">
        <xs:complexType mixed="true">
            <xs:attribute name="phoneme" type="xs:string" use="optional"/>
        </xs:complexType>
    </xs:element>

    <!-- See full schema file for complete definition -->

</xs:schema>

Usage Example

SAML Document Example 
<?xml version="1.0" encoding="UTF-8"?>
<speak xmlns="http://www.w3.org/2001/10/synthesis"
       xmlns:saml="https://shunyalabs.ai/2026/saml"
       version="1.1"
       xml:lang="en-US">

    <saml:metadata>
        <saml:file id="recording-001" task="reading"/>
        <saml:speaker id="SPK01" age-group="adult"/>
    </saml:metadata>

    <p>I <saml:repetition type="incomplete-syllable" 
        severity="2" count="2">
        <saml:unit>w</saml:unit>
        <saml:unit>w</saml:unit>
    </saml:repetition>was going to the store.</p>

</speak>
